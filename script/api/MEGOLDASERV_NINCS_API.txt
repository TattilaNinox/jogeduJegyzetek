3.2. Véletlenszerű Jogeset Lekérése
------------------------------------
Flutter kód példa:

```dart
import 'package:cloud_firestore/cloud_firestore.dart';
import 'dart:math';

Future<Map<String, dynamic>?> getRandomJogeset(String paragrafus) async {
  try {
    // Firestore-ból lekérdezés
    DocumentSnapshot doc = await FirebaseFirestore.instance
        .collection('jogesetek')
        .doc(paragrafus.replaceAll(':', '_'))
        .get();
    
    if (!doc.exists) {
      return null; // Nincs ilyen paragrafus
    }
    
    Map<String, dynamic> data = doc.data() as Map<String, dynamic>;
    List<dynamic> jogesetek = data['jogesetek'] as List<dynamic>;
    
    if (jogesetek.isEmpty) {
      return null; // Nincs jogeset
    }
    
    // Véletlenszerű választás
    Random random = Random();
    int randomIndex = random.nextInt(jogesetek.length);
    
    return jogesetek[randomIndex] as Map<String, dynamic>;
  } catch (e) {
    print('Hiba: $e');
    return null;
  }
}
```

5.1. Gemini 3 Pro Integráció
-----------------------------
Függőségek:
- google-generativeai
- firebase-admin

Kód struktúra:
```python
import google.generativeai as genai
from firebase_admin import firestore

# Gemini 3 Pro beállítás
genai.configure(api_key="GEMINI_API_KEY")
model = genai.GenerativeModel('gemini-3-pro')

# Generálás
def generate_jogeset(paragrafus_szoveg, paragrafus_szam):
    prompt = f"""
    Generálj egy valósághű jogesetet a következő jogszabály alapján:
    
    {paragrafus_szoveg}
    
    Paragrafus: {paragrafus_szam}
    
    A jogeset tartalmazza:
    - Cím
    - Tények (részletes, valósághű)
    - Kérdés
    - Megoldás (jogszabály alkalmazása)
    - Komplexitás (egyszerű/közepes/komplex)
    """
    
    response = model.generate_content(prompt)
    # JSON formátumba konvertálás
    return parse_response(response)
```

5.2. Batch Generálás
--------------------
```python
def generate_multiple_jogesetek(paragrafus, darabszam):
    jogesetek = []
    for i in range(darabszam):
        jogeset = generate_jogeset(paragrafus_szoveg, paragrafus)
        jogeset['id'] = i + 1
        jogesetek.append(jogeset)
    
    # Firestore-ba mentés
    db = firestore.client()
    doc_ref = db.collection('jogesetek').document(paragrafus.replace(':', '_'))
    doc_ref.set({
        'paragrafus': f'{paragrafus}. §',
        'kategoria': determine_category(paragrafus),
        'jogesetek': jogesetek
    })
```
